  X, y = make_moons(n_samples=200,
                    shuffle=True,
                    noise=0.5,
                    random_state=10)
  ratio = 0.1
  X_, y_ = make_imbalance(X, y, ratio=ratio, min_c_=1)
 
  #fig = plt.figure()
  #ax = fig.add_subplot(111)
  plt.gcf()
  plt.plot(X_[y_ == 0, 0],
           X_[y_ == 0, 1],
           label="Class #0",
           alpha=0.5,
           edgecolor=almost_black,
           facecolor=palette[0],
           linewidth=0.15)

  plt.plot(X_[y_ == 1, 0],
           X_[y_ == 1, 1],
           label="Class #1",
